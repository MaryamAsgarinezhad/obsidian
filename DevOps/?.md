        -- Remove the cache file  
        local lfs = require("lfs")  
        local attr = lfs.attributes(cache_file)  
        if attr and attr.mode == "file" then  
            os.remove(cache_file)

        else  
            ngx.say("File not found: " .. cache_file)  
        end  
        return ngx.exit(200)
